% ── SLIDE 2: Motivation I ─────────────────────────────────────────────────────
\begin{frame}{Motivation I: The Data Center Bandwidth Challenge}
  \begin{columns}[T]
    \column{0.56\textwidth}
      \textbf{Global data-center traffic is exploding}
      \begin{itemize}\setlength\itemsep{3pt}
        \item Reached \textbf{20.6 ZB/year} in 2021, growing $\sim$25\%/yr \footnotemark[1]
        \item 77\% of traffic stays \emph{inside} data centers
        \item Copper wires hit fundamental limits:\\
              {\small resistance heat, EM crosstalk, energy-per-bit ceiling}
      \end{itemize}
      \vspace{0.5em}
      \textbf{Optical interconnects are the only path forward}
      \begin{itemize}\setlength\itemsep{3pt}
        \item 400GbE / 800GbE standards mandate WDM optics \footnotemark[2]
        \item \textbf{WDM}: pack $N$ color channels onto one fiber
        \item Receiver must \textbf{demultiplex} — separate each color
      \end{itemize}
    \column{0.41\textwidth}
      \begin{block}{What a demultiplexer does}
        \centering
        \begin{tikzpicture}[scale=0.82]

% ── SLIDE 3: Motivation II ────────────────────────────────────────────────────
\begin{frame}{Motivation II: Why an Arrayed Waveguide Grating on Silicon?}
  \begin{columns}[T]
    \column{0.50\textwidth}
      \textbf{AWG: industry-standard on-chip demux} \footnotemark[3]
      \begin{itemize}\setlength\itemsep{3pt}
        \item First proposed by Smit (1988); now in every coherent transceiver
        \item Channels: 4–96; insertion loss $<$2\,dB in production
        \item Works as a \emph{phased-array spatial filter} — no moving parts
      \end{itemize}
      \vspace{0.4em}
      \textbf{Why silicon (SOI platform)?} \footnotemark[4]
      \begin{itemize}\setlength\itemsep{3pt}
        \item CMOS-fab compatible — made in standard chip factories
        \item Huge index contrast Si/SiO$_2$ ($\Delta n \approx 2.0$)\\
              $\Rightarrow$ bend radius $<$10\,$\mu$m, chip area $\sim$mm$^2$
        \item Co-integrates with Ge detectors \& Si modulators
      \end{itemize}
    \column{0.47\textwidth}
      \begin{block}{Our 4-channel C-band target (ITU-T G.694.1 \footnotemark[5])}
        \centering\small
        \renewcommand{\arraystretch}{1.15}
        \begin{tabular}{cc}
          \toprule
          Channel & Center $\lambda$ (nm) \\
          \midrule
          ch\,1 & 1547.6 \\
          ch\,2 & 1549.2 \\
          ch\,3 & 1550.8 \\
          ch\,4 & 1552.4 \\
          \midrule
          \multicolumn{2}{c}{Spacing: 1.6\,nm $\approx$ 200\,GHz (DWDM)}\\
          \bottomrule
        \end{tabular}
      \end{block}
      \vspace{0.3em}
      \small Platform: 500\,nm\,$\times$\,220\,nm Si strip waveguide on SiO$_2$.
  \end{columns}
  \footnotetext[3]{\tiny M.~K.~Smit \& C.~van Dam, \textit{IEEE JSTQE} 2, 236--250 (1996).}
  \footnotetext[4]{\tiny L.~Chrostowski \& M.~Hochberg, \textit{Silicon Photonics Design}, Cambridge, 2015.}
  \footnotetext[5]{\tiny ITU-T G.694.1, DWDM frequency grid, 2012.}
\end{frame}

% ── SLIDE 10: Simplified Spectrum Model ──────────────────────────────────────
\begin{frame}{Phase 3a — Array-Factor Spectrum Model}
  \begin{columns}[T]
    \column{0.43\textwidth}
      \textbf{Analytic formula (zero simulation cost):}
      \[
        T(\lambda) = T_\text{pk}\!
        \underbrace{\left(\frac{\sin(N\phi/2)}{N\sin(\phi/2)}\right)^{\!2}}_{\text{array factor}}
        \!\cdot\!
        \underbrace{e^{-\delta\lambda^2/2\sigma^2}}_{\text{envelope}}
      \]
      $\phi\!=\!2\pi\delta\lambda/\text{FSR}$,\;$N\!=\!16$,\;$\sigma\!=\!6$\,nm

      \vspace{0.5em}
      \begin{block}{\small Predicted channel metrics}
        \small
        \begin{tabular}{ll}
          Insertion loss & 3.0\,dB \\
          Crosstalk      & $-13.7$\,dB \\
          3\,dB BW       & 1.06\,nm \\
          Ch.\ spacing   & 1.60\,nm\;\checkmark \\
        \end{tabular}
      \end{block}
      \begin{alertblock}{\small Model limitation}
        \small Symmetric model — all 4 channels look identical.\\
        Real device: edge channels have higher IL.
      \end{alertblock}

\begin{frame}{Phase 3b — 2D FDTD Validation Setup \& Quality Rubric}
  \begin{columns}[T]
    \column{0.50\textwidth}
      \textbf{Why 2D, not full 3D?}
      \begin{itemize}\small\setlength\itemsep{3pt}
        \item Full 3D device ($\sim$2\,mm long) would be prohibitively expensive
        \item \textbf{2D effective-index trick:} replace the 220\,nm Si slab with a
              2D medium at $n\!=\!n_\text{eff}$ — same in-plane physics, much smaller grid
        \item Only the output-FPR section is simulated; arm phases are injected analytically
        \item Saves $\sim$100$\times$ compared to full 3D
      \end{itemize}
